#include<stdio.h>
int main()
{
int a[100],i,n;
printf("enter n value");
scanf("%d",&n);
printf("enter elements\n");
for(i=0;i<n;i++)
{
scanf("%d",&aa[i]);
}

int display(int a[],int start,int end)
{
if(start>end)
return;
else
printf("%d\n",a[start]);
display(a,start+1,end);
}
